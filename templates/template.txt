	${SEL_STATEMENTS}
    ,${REC_ID} AS RCORD_ID
	,'I' AS REC_IND
	,SESS.EXECUTION_SESSION_ID 	
	,SESS.EXECUTION_SESSION_ID

	(
		SELECT * FROM
		
		(
			SELECT ${SRC_COLUMNS}
			,'No Info in SMX' AS SRC_TAB_KEY,'${SRC_TABLE}' AS SRC_TAB_NAME,	'${SRC_COLUMNS}' AS SRC_COL_NAME
			FROM &&ENV_STAGE.${SRC_TABLE} SRC 
			${WHERE_COND}

	) SRC
